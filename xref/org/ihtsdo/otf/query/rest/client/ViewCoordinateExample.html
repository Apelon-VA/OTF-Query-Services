35      public static SimpleViewCoordinate getSnomedInferredLatest() throws JAXBException {
36          
37          
38          SimpleViewCoordinate svc = new SimpleViewCoordinate();
39          svc.setName("Snomed Inferred Latest");
40          svc.setClassifierSpecification(getSpec("IHTSDO Classifier", 
41                  "7e87cc5b-e85f-3860-99eb-7a44f2b9e6f9"));
42          svc.setLanguageSpecification(getSpec("United States of America English language reference set (foundation metadata concept)", 
43                  "bca0a686-3516-3daf-8fcf-fe396d13cfad"));
44          svc.getLanguagePreferenceOrderList().add(svc.getLanguageSpecification());
45          svc.getAllowedStatus().add(Status.ACTIVE);
46          svc.setPrecedence(Precedence.PATH);
47          SimplePath wbAuxPath = new SimplePath();
48          wbAuxPath.setPathConceptSpecification(getSpec("Workbench Auxiliary", 
49                  "2faa9260-8fb2-11db-b606-0800200c9a66"));
50          SimplePosition snomedWbAuxOrigin = new SimplePosition();
51          snomedWbAuxOrigin.setPath(wbAuxPath);
52          // Long.MAX_VALUE == latest
53          snomedWbAuxOrigin.setTimePoint(Long.MAX_VALUE);
54          
55          SimplePath snomedCorePath = new SimplePath();
56          snomedCorePath.setPathConceptSpecification(getSpec("SNOMED Core", 
57                  "8c230474-9f11-30ce-9cad-185a96fd03a2"));
58          snomedCorePath.getOrigins().add(snomedWbAuxOrigin);
59          
60          SimplePosition latestOnSnomedPath = new SimplePosition();
61          latestOnSnomedPath.setPath(snomedCorePath);
62          latestOnSnomedPath.setTimePoint(Long.MAX_VALUE);
63          svc.setViewPosition(latestOnSnomedPath);
64          
65          svc.setRelAssertionType(RelAssertionType.INFERRED);
66          svc.setContradictionPolicy(ContradictionManagerPolicy.LAST_COMMIT_WINS);
67          svc.setLangSort(LanguageSort.RF_2_LANG_REFEX);
68          
69          
70          return svc;
71  
72      }
73  
74      private static SimpleConceptSpecification getSpec(String description, String uuidStr) {
75          SimpleConceptSpecification classifierSpec = new SimpleConceptSpecification();
76          classifierSpec.setDescription(description);
77          classifierSpec.setUuid(uuidStr);
78          return classifierSpec;
79      }
