184     /**
185      *
186      * @param query The query to apply.
187      * @param field The component field to be queried.
188      * @param sizeLimit The maximum size of the result list.
189      * @param targetGeneration target generation that must be included in the
190      * search or Long.MIN_VALUE if there is no need to wait for a target
191      * generation.
192      * @return a List of <code>SearchResult</codes> that contins the nid of the
193      * component that matched, and the score of that match relative to other
194      * matches.
195      * @throws IOException
196      * @throws ParseException
197      */
198     public final List<SearchResult> query(String query, ComponentProperty field, int sizeLimit, long targetGeneration)
199             throws IOException, ParseException {
200         try {
201             List<SearchResult> result;
202 
203             switch (field) {
204                 case LONG_EXTENSION_1:
205                     long long1 = Long.parseLong(query);
206                     Query long1query = NumericRangeQuery.newLongRange(field.name(), long1, long1, true, true);
207 
208                     result = search(long1query, sizeLimit, targetGeneration);
209 
210                     break;
211 
212                 case DESCRIPTION_TEXT:
213                     Query q = new QueryParser(LuceneIndexer.luceneVersion, field.name(),
214                             new StandardAnalyzer(LuceneIndexer.luceneVersion)).parse(query);
215 
216                     result = search(q, sizeLimit, targetGeneration);
217 
218                     if (result.size() > 0) {
219                         if (TermstoreLogger.logger.isLoggable(Level.FINE)) {
220                             TermstoreLogger.logger.log(Level.FINE, "StandardAnalyzer query returned {0} hits",
221                                     result.size());
222                         }
223                     } else {
224                         if (TermstoreLogger.logger.isLoggable(Level.FINE)) {
225                             TermstoreLogger.logger.fine("StandardAnalyzer query returned no results. "
226                                     + "Now trying WhitespaceAnalyzer query");
227                             q = new QueryParser(LuceneIndexer.luceneVersion, field.name(),
228                                     new WhitespaceAnalyzer(LuceneIndexer.luceneVersion)).parse(query);
229                         }
230 
231                         result = search(q, sizeLimit, targetGeneration);
232                     }
233 
234                     break;
235 
236                 default:
237                     throw new IOException("Can't handle: " + field.name());
238             }
239 
240             return result;
241         } catch (ParseException | IOException | NumberFormatException e) {
242             throw new IOException(e);
243         }
244     }

282     @Override
283     public final Future<Long> index(ComponentChronicleBI chronicle) {
284         if (!enabled) {
285             return null;
286         }
287 
288         if (indexChronicle(chronicle)) {
289             Future<Long> future = luceneWriterService.submit(new AddDocument(chronicle));
290 
291             luceneWriterFutureCheckerService.execute(new FutureChecker(future));
292 
293             return future;
294         }
295 
296         return unindexedFuture;
297     }

299     private List<SearchResult> search(Query q, int sizeLimit, long targetGeneration) throws IOException {
300         if (targetGeneration != Long.MIN_VALUE) {
301             searcherManager.waitForGeneration(targetGeneration);
302         }
303 
304         IndexSearcher searcher = searcherManager.acquire();
305 
306         try {
307             TopDocs topDocs = searcher.search(q, sizeLimit);
308             List<SearchResult> results = new ArrayList<>(topDocs.totalHits);
309 
310             for (ScoreDoc hit : topDocs.scoreDocs) {
311                 if (TermstoreLogger.logger.isLoggable(Level.FINE)) {
312                     TermstoreLogger.logger.log(Level.FINE, "Hit: {0} Score: {1}", new Object[]{hit.doc, hit.score});
313                 }
314 
315                 Document doc = searcher.doc(hit.doc);
316 
317                 results.add(
318                         new SearchResult(
319                                 doc.getField(ComponentProperty.COMPONENT_ID.name()).numericValue().intValue(), hit.score));
320             }
321 
322             return results;
323         } finally {
324             searcherManager.release(searcher);
325         }
326     }

331      * document is added to the index.
332      * @return a <code>Callable&lt;Long&gt;</code> object that will block until

339     public IndexedGenerationCallable getIndexedGenerationCallable(int nid) {
340         IndexedGenerationCallable indexedLatch = new IndexedGenerationCallable();
341         IndexedGenerationCallable existingIndexedLatch = componentNidLatch.putIfAbsent(nid, indexedLatch);
342 
343         if (existingIndexedLatch != null) {
344             return existingIndexedLatch;
345         }
346 
347         return indexedLatch;
348     }

360     private class AddDocument implements Callable<Long> {
361 
362         ComponentChronicleBI chronicle;
363 
364         public AddDocument(ComponentChronicleBI chronicle) {
365             this.chronicle = chronicle;
366         }
367 
368         @Override
369         public Long call() throws Exception {
370             IndexedGenerationCallable latch = componentNidLatch.remove(chronicle.getNid());
371             Document doc = new Document();
372 
373             doc.add(new IntField(ComponentProperty.COMPONENT_ID.name(), chronicle.getNid(),
374                     LuceneIndexer.indexedComponentNidType));
375             addFields(chronicle, doc);
376 
377             // Note that the addDocument operation could cause duplicate documents to be
378             // added to the index if a new luceneVersion is added after initial index
379             // creation. It does this to avoid the performance penalty of
380             // finding and deleting documents prior to inserting a new one.
381             //
382             // At this point, the number of duplicates should be
383             // small, and we are willing to accept a small number of duplicates
384             // because the new versions are additive (we don't allow deletion of content)
385             // so the search results will be the same. Duplicates can be removed
386             // by regenerating the index.
387             long indexGeneration = trackingIndexWriter.addDocument(doc);
388 
389             if (latch != null) {
390                 latch.setIndexGeneration(indexGeneration);
391             }
392 
393             return indexGeneration;
394         }
