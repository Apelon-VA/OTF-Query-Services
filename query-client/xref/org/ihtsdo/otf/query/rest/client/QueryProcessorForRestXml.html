52      public static String process(SimpleViewCoordinate viewpoint, 
53              ForCollection forObject,
54              LetMap letMap,
55              Where where,
56              ReturnTypes returnType,
57              String host) throws JAXBException, IOException {
58  
59          
60          
61          // create the client
62          Client client = ClientBuilder.newClient();
63          // specify the host and the path. 
64          WebTarget target = client.target(host).path("query-service/query");
65          
66          
67          return target.queryParam("VIEWPOINT", getXmlString(viewpoint)).
68                        queryParam("FOR", getXmlString(forObject)).
69                        queryParam("LET", getXmlString(letMap)).
70                        queryParam("WHERE", getXmlString(where)).
71                        queryParam("RETURN", getXmlString(returnType)).
72                  request(MediaType.TEXT_PLAIN).get(String.class);
73  
74      }
75  
76      private static String getXmlString(Object obj) throws JAXBException {
77          if (obj instanceof SimpleViewCoordinate) {
78               org.ihtsdo.otf.jaxb.chronicle.api.ObjectFactory factory = new org.ihtsdo.otf.jaxb.chronicle.api.ObjectFactory();
79              obj =  factory.createSimpleViewCoordinate((SimpleViewCoordinate) obj);
80  
81          } else if (obj instanceof ForCollection) {
82              org.ihtsdo.otf.jaxb.query.ObjectFactory factory = new org.ihtsdo.otf.jaxb.query.ObjectFactory();
83              obj = factory.createForCollection((ForCollection) obj);
84          } else if (obj instanceof LetMap) {
85              org.ihtsdo.otf.jaxb.query.ObjectFactory factory = new org.ihtsdo.otf.jaxb.query.ObjectFactory();
86              obj = factory.createLetMap((LetMap) obj);
87          } else if (obj instanceof Where) {
88              org.ihtsdo.otf.jaxb.query.ObjectFactory factory = new org.ihtsdo.otf.jaxb.query.ObjectFactory();
89              obj = factory.createWhere((Where) obj);
90          } else if (obj instanceof ReturnTypes) {
91              org.ihtsdo.otf.jaxb.query.ObjectFactory factory = new org.ihtsdo.otf.jaxb.query.ObjectFactory();
92              obj = factory.createReturnTypes((ReturnTypes) obj);
93          }
94          StringWriter writer = new StringWriter();
95          ctx.createMarshaller().marshal(obj, writer);
96          return writer.toString();
97      }
