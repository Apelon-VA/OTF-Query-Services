16  package org.ihtsdo.otf.query.rest.client.examples;
17  
18  import javax.ws.rs.client.Client;
19  import javax.ws.rs.client.ClientBuilder;
20  import javax.ws.rs.client.WebTarget;
21  import javax.ws.rs.core.MediaType;
22  import javax.ws.rs.core.Response;
23  
24  /**
25   * This example validates a simple connection to the query service via a Apache
26   * Jersey REST client. This simple example is intended to demonstrate proper
27   * connectivity and service availability. This example takes advantage of a
28   * hello service on the server that is just there for simple verification. When
29   * a path of "query-service/hello/frank" is provided, it will return "hello
30   * frank." Similarly, if "query-service/hello/bob" is provided, it will return
31   * "hello bob."
32   *
33   * @author kec
34   */
35  public class HelloExample {
36  
37      /**
38       * 
39       * @param args args[0] is an optional server url. 
40       */
41      public static void main(String[] args) {
42          // default host.
43          String host = "http://api.snomedtools.com/";
44          
45          // if host is provided, override default host.
46          if (args.length > 0) {
47              host = args[0];
48          }
49          // create the client
50          Client client = ClientBuilder.newClient();
51          // specify the host and the path. 
52          WebTarget target = client.target(host).path("query-service/hello/frank");
53          // get the response from the server
54          Response response = target.request(MediaType.TEXT_PLAIN_TYPE).get();
55          // should return "200"
56          System.out.println(response.getStatus()); 
57          // should return "hello frank."
58          System.out.println(response.readEntity(String.class)); 
59      }
60  }
